<script>
const API_URL = 'https://109.248.162.180:8080';
const CREATE_TOKEN = '9afd436e-064e';

let pendingEmail = '';
let sessionToken = localStorage.getItem('session_token');
let selectedServer = null;

if (sessionToken) {
    fetch(`${API_URL}/api/verify-session`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ token: sessionToken })
    })
    .then(r => r.json())
    .then(data => {
        if (data.ok) {
            showServerSection();
            document.getElementById('user-email-display').textContent = data.email;
            document.getElementById('user-email-display').classList.remove('hidden');
            document.getElementById('logout-btn').classList.remove('hidden');
        } else {
            localStorage.removeItem('session_token');
        }
    });
}

// OTP inputs
const digits = document.querySelectorAll('.digit-input');
digits.forEach((digit, index) => {
    digit.addEventListener('input', (e) => {
        if (e.target.value.length === 1 && index < 5) digits[index + 1].focus();
    });
    digit.addEventListener('keydown', (e) => {
        if (e.key === 'Backspace' && !e.target.value && index > 0) digits[index - 1].focus();
    });
});

function getOtpCode() {
    let code = '';
    digits.forEach(d => code += d.value);
    return code;
}

// Send code
document.getElementById('send-code-btn')?.addEventListener('click', async () => {
    const email = document.getElementById('login-email').value.trim();
    const errorEl = document.getElementById('email-error');
    if (!email) {
        errorEl.textContent = 'Введите email';
        errorEl.classList.remove('hidden');
        return;
    }
    errorEl.classList.add('hidden');
    const btn = document.getElementById('send-code-btn');
    btn.disabled = true;
    btn.textContent = 'Отправка...';
    try {
        const res = await fetch(`${API_URL}/api/send-code`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ email })
        });
        const data = await res.json();
        if (!res.ok) throw new Error(data.error);
        pendingEmail = email;
        document.getElementById('otp-email').textContent = email;
        document.getElementById('email-section').classList.add('hidden');
        document.getElementById('otp-section').classList.remove('hidden');
        digits.forEach(d => d.value = '');
        digits[0].focus();
    } catch (e) {
        errorEl.textContent = e.message;
        errorEl.classList.remove('hidden');
    } finally {
        btn.disabled = false;
        btn.textContent = 'Отправить код';
    }
});

// Verify code
document.getElementById('verify-code-btn')?.addEventListener('click', async () => {
    const code = getOtpCode();
    const errorEl = document.getElementById('otp-error');
    if (code.length !== 6) {
        errorEl.textContent = 'Введите 6-значный код';
        errorEl.classList.remove('hidden');
        return;
    }
    errorEl.classList.add('hidden');
    const btn = document.getElementById('verify-code-btn');
    btn.disabled = true;
    btn.textContent = 'Проверка...';
    try {
        const res = await fetch(`${API_URL}/api/verify-code`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ email: pendingEmail, code })
        });
        const data = await res.json();
        if (!res.ok) throw new Error(data.error);
        localStorage.setItem('session_token', data.token);
        document.getElementById('otp-section').classList.add('hidden');
        showServerSection();
        document.getElementById('user-email-display').textContent = pendingEmail;
        document.getElementById('user-email-display').classList.remove('hidden');
        document.getElementById('logout-btn').classList.remove('hidden');
    } catch (e) {
        errorEl.textContent = e.message;
        errorEl.classList.remove('hidden');
    } finally {
        btn.disabled = false;
        btn.textContent = 'Подтвердить';
    }
});

// Resend code
document.getElementById('resend-code-btn')?.addEventListener('click', async () => {
    await fetch(`${API_URL}/api/send-code`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ email: pendingEmail })
    });
});

// Logout
document.getElementById('logout-btn')?.addEventListener('click', () => {
    localStorage.removeItem('session_token');
    location.reload();
});

// Load servers
fetch(`${API_URL}/api/servers`)
    .then(r => r.json())
    .then(servers => {
        const container = document.getElementById('server-list');
        for (const [key, s] of Object.entries(servers)) {
            const btn = document.createElement('button');
            btn.className = 'server-btn bg-surface-container-high p-4 rounded-xl text-center transition-all hover:scale-105';
            btn.innerHTML = `<span class="text-3xl mb-2 block">${s.flag}</span><span class="font-semibold">${s.name}</span>`;
            btn.onclick = () => {
                document.querySelectorAll('.server-btn').forEach(b => b.classList.remove('selected'));
                btn.classList.add('selected');
                selectedServer = key;
            };
            container.appendChild(btn);
            if (!selectedServer) btn.click();
        }
    });

function showServerSection() {
    document.getElementById('server-section').classList.remove('hidden');
    document.getElementById('result-section').classList.add('hidden');
    window.scrollTo({ top: document.getElementById('server-section').offsetTop - 80, behavior: 'smooth' });
}

function showResult(data) {
    document.getElementById('server-section').classList.add('hidden');
    document.getElementById('result-section').classList.remove('hidden');
    document.getElementById('qr-img').src = 'data:image/png;base64,' + data.qr;
    document.getElementById('vless-url').textContent = data.vless_url;
    document.getElementById('sub-url').textContent = data.sub_url;
    document.getElementById('result-info').textContent = `${data.username || pendingEmail} — ${data.server}`;
    window.scrollTo({ top: document.getElementById('result-section').offsetTop - 80, behavior: 'smooth' });
}

async function createKey() {
    const username = document.getElementById('username').value.trim() || pendingEmail.split('@')[0];
    const errorEl = document.getElementById('create-error');
    if (!selectedServer) {
        errorEl.textContent = 'Выберите сервер';
        errorEl.classList.remove('hidden');
        return;
    }
    errorEl.classList.add('hidden');
    const btn = document.getElementById('create-key-btn');
    btn.disabled = true;
    btn.textContent = 'Генерация...';
    try {
        const res = await fetch(`${API_URL}/api/create`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ server: selectedServer, username, token: CREATE_TOKEN })
        });
        const data = await res.json();
        if (!res.ok) throw new Error(data.error);
        showResult(data);
    } catch (e) {
        errorEl.textContent = e.message;
        errorEl.classList.remove('hidden');
    } finally {
        btn.disabled = false;
        btn.textContent = 'Сгенерировать ключ';
    }
}

function resetForm() {
    document.getElementById('username').value = '';
    document.getElementById('server-section').classList.remove('hidden');
    document.getElementById('result-section').classList.add('hidden');
}

function copyText(elementId, btn) {
    const text = document.getElementById(elementId).textContent;
    navigator.clipboard.writeText(text).then(() => {
        btn.classList.add('copied');
        btn.textContent = 'Скопировано!';
        setTimeout(() => {
            btn.classList.remove('copied');
            btn.textContent = 'Скопировать';
        }, 2000);
    });
}

document.getElementById('create-key-btn')?.addEventListener('click', createKey);
document.getElementById('reset-btn')?.addEventListener('click', resetForm);
document.getElementById('login-email')?.addEventListener('keypress', (e) => {
    if (e.key === 'Enter') document.getElementById('send-code-btn').click();
});
</script>
